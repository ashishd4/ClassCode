#ifndef ILLINI_BOOK_HPP
#define ILLINI_BOOK_HPP

#include <string>
#include <vector>
#include <map>
#include <queue>

class IlliniBook {

  size_t CountGroups(const std::vector<std::string> &relationships) const;

private:

  std::map<int, std::map<int, std::string>> ships_;
};

#endif
