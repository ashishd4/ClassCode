#include "color.hpp"

#include <fstream>
#include <iostream>

using namespace std;

Color::Color(int r, int g, int b): red_(r), green_(g), blue_(b) {
  // need to ensure valid color has been constructed...
  if (r < 0 || r > kMax || b < 0 || b > kMax || g < 0 || g > kMax) {
    cout << "Red:  " << r << endl;
    cout << "Green:  " << b << endl;
    cout << "Blue:  " << g << endl;
    throw runtime_error("Invalid color input");
  }
}

// do not modify
bool operator==(const Color& rhs, const Color& lhs) {
  return (rhs.Red() == lhs.Red() && rhs.Green() == lhs.Green() &&
          rhs.Blue() == lhs.Blue());
}
