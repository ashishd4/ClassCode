#include <iostream>

#include "functions.hpp"
#include "utilities.hpp"
using namespace std;

int main(int argc, char* argv[]) {
  if (argc != 3) {
    cerr << "Usage: " << argv[0] << " [input_file] [dna_sequence]" << std::endl;
    return 1;
  }

  string sequence = argv[2];

  vector<vector<string>> table;

  ifstream ifs{argv[1]};
  for (string line; getline(ifs, line); line = "")
    table.push_back(utilities::GetSubstrs(line, ','));  // Formed 2d vector

  string name = FindMatchingDNA(table, sequence);
  cout << name << endl;
}
