#include <iostream>
#include <vector>

#include "dna_strand.hpp"

int main() {
    DNAstrand test;
    std::vector<char> v;

    v.push_back('c');
    v.push_back('t');
    v.push_back('a');
    v.push_back('t');
    v.push_back('a');
    Node* prev = new Node(v[0]);
    Node* head = prev;
    for (size_t i = 1; i < v.size(); i++) {
        Node* node = new Node(v.at(i));
        prev->next = node;
        prev = node;
    }
    char letters[] = "ct";
    test.assignHead(head);
    test.assignTail(prev);
    DNAstrand repl;
    std::vector<char> replace;

    replace.push_back('t');
    replace.push_back('g');
    replace.push_back('a');
    prev = new Node(replace[0]);
    head = prev;
    for (size_t i = 1; i < replace.size(); i++) {
        Node* node = new Node(replace.at(i));
        prev->next = node;
        prev = node;
    }

    repl.assignHead(head);
    repl.assignTail(prev);
    test.SpliceIn(letters, repl);
}
